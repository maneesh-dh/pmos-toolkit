    _selftest_awk_lint() {
  # Scan every awk '...' block body for word-boundary atoms.
  # Returns 0 if clean, 1 if any \b, \< or \> found.
  local script="$1"
  local lint_failed=0
  # Use grep -nE to find lines that look like inside an awk single-quoted block
  # carrying \b, \<, or \>. Heuristic: awk patterns are quoted with ' on the
  # same line as the `awk` keyword. We grep for `awk '...\\b...'` style.
  # Two-pass scan (per /plan review loop 1 — Recommended).
  # Pass A: inline single-line awk blocks: `awk '...\b...'` on one line.
  while IFS= read -r match; do
    readme::log "[lint] FAIL: \\b at $match"
    lint_failed=1
  done < <(grep -nE "awk[[:space:]]+'[^']*\\\\(b|<|>)" "$script" || true)
  # Pass B: multi-line awk blocks. Track awk-scope across lines via awk itself:
  # opening single quote on a line containing `awk`, closing single quote later.
  # Any `\b`/`\<`/`\>` between is a finding.
  while IFS= read -r match; do
    readme::log "[lint] FAIL: \\b at $match (multi-line awk block)"
    lint_failed=1
  done < <(awk '
    /awk[[:space:]]+'\''/ && !/'\''/{in_awk=1; awk_line=NR; next}
    in_awk && /'\''/{in_awk=0; next}
    in_awk && /\\b|\\<|\\>/{ printf "%s:%d\n", FILENAME, NR }
  ' "$script" || true)
  return $lint_failed
}
